#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main(void)
{
    int max = 0;
    int a[9];
    int b = 0;
    for (int i = 1; i <= 9; i++)
    {
        scanf("%d", &a[i]);
        if (a[i] > max)
        {
            max = a[i];
            b = i;
        }
    }
    printf("%d\n%d", max, b);
    return 0;
}

#include <stdio.h>
int main(void)
{
    int arr[256];
	int n = 0; // 주어진 정수의 개수
	int v = 0; // 찾을 정수
	int count = 0; // 찾을 정수의 개수
    scanf("%d", &n);
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &arr[i]);
	}
    scanf("%d", &v);
    for (int j = 0; j < n; j++) 
    {
        if (v == arr[j])
            count++;
	}
    printf("%d", count);
	return 0;
}
